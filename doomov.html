const PROXY = '/api/proxy-k24hd?url='; // 1. เปลี่ยนโครงสร้างหมวดหมู่หลักวิ่งไปหา "หมวดหมู่หนังใหม่" const BASE_CAT = 'https://kubhd24.net/category/movie-new/'; async function request(url, isJson = false) { try { const res = await fetch(PROXY + encodeURIComponent(url)); if (!res.ok) return null; return isJson ? await res.json() : await res.text(); } catch (e) { return null; } } async function getStream(postId) { const json = await request(`https://kubhd24.net/wp-admin/admin-ajax.php?action=mix_get_player&post_id=${postId}`, true); if (!json?.success) return null; const match = json.player.match(/data-src="([^"]+)"/); if (!match) return null; return `https://media.vdohls.com/${match[1].split('/').pop()}/playlist.m3u8`; } // 2. ปรับการ parse ข้อมูลให้วิ่งไปที่หน้าหนังเดี่ยว async function parseMovie(id) { const html = await request(`https://kubhd24.net/movie/${id}/`); if (!html) return null; const doc = new DOMParser().parseFromString(html, "text/html"); const title = doc.querySelector("h1")?.textContent.trim() || id; const poster = doc.querySelector("img.wp-post-image")?.getAttribute("data-src") || doc.querySelector("img.wp-post-image")?.src; // ค้นหา Post ID ของหนังเรื่องนี้ เพื่อนำไปขอลิงก์ Player หลังบ้าน const playerDiv = doc.querySelector("div.player-wrapper") || doc.querySelector("[id*='player']"); // มองหาปุ่มหรือคีย์ของโพสต์หนังเดี่ยว const favoriteButton = doc.querySelector("button.add-favorite"); const postId = favoriteButton ? favoriteButton.getAttribute("data-post-id") : null; let episodes = []; if (postId) { const url = await getStream(postId); if (url) { episodes.push({ title: "Full Movie", url }); } } else { // แผนสำรองหาปุ่มเล่นถ้ามี const playBtn = doc.querySelector("button.ep, .play-btn"); if (playBtn && playBtn.id) { const url = await getStream(playBtn.id); if (url) episodes.push({ title: "Full Movie", url }); } } return { title, poster, episodes }; } async function startHarvesting() { const startPage = parseInt(document.getElementById("startPage").value); const endPage = parseInt(document.getElementById("endPage").value); const loader = document.getElementById("loader"); const output = document.getElementById("output"); const stats = document.getElementById("stats"); output.innerHTML = ""; let totalMoviesFound = 0; for (let p = startPage; p <= endPage; p++) { const pageUrl = p === 1 ? BASE_CAT : `${BASE_CAT}page/${p}/`; loader.textContent = `🔍 กำลังสแกนหน้าหมวดหมู่หนังที่ ${p}...`; const catHtml = await request(pageUrl); if (!catHtml) continue; // 3. ปรับ Regex ค้นหา ให้จับเฉพาะเส้นทางที่เป็นหนังภาพยนตร์ /movie/ const regex = /href="https:\/\/kubhd24\.net\/movie\/([^"\/]+)\//g; const ids = Array.from(new Set([...catHtml.matchAll(regex)].map(m => m[1]))).filter(i => i !== 'movie'); totalMoviesFound += ids.length; stats.textContent = `กำลังจัดการหน้าที่ ${p} (พบหนังใหม่ ${ids.length} เรื่อง / รวมสะสม ${totalMoviesFound} เรื่อง)`; for (let i = 0; i < ids.length; i++) { loader.textContent = `⏳ หน้า ${p} | เรื่องที่ ${i+1}/${ids.length}: ${ids[i]}`; const data = await parseMovie(ids[i]); if (data && data.episodes.length > 0) { renderCard(data, p); } } } loader.textContent = `✅ ดึงข้อมูลหนังสำเร็จ! รวมทั้งหมด ${totalMoviesFound} เรื่อง จากหน้า ${startPage}-${endPage}`; } function renderCard(data, pageNum) { let m3u = "#EXTM3U\n"; data.episodes.forEach(ep => { m3u += `#EXTINF:-1 tvg-logo="${data.poster}" group-title="ภาพยนตร์เดี่ยว", ${data.title}\n${ep.url}\n`; }); const card = document.createElement("div"); card.className = "card"; card.innerHTML = `
หน้า ${pageNum}
${data.title}
ภาพยนตร์เต็มเรื่อง
${m3u}
`; document.getElementById("output").appendChild(card); } function toggleM3U(btn) { const pre = btn.nextElementSibling; pre.style.display = (pre.style.display === 'block') ? 'none' : 'block'; btn.textContent = (pre.style.display === 'block') ? '✖ ปิด' : '📂 แสดง M3U Playlist'; }
